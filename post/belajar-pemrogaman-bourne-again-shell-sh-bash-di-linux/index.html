  4. Penulisan Kode

    #!/bin/bash

    #------------------------------------#
    #Membuat Script Bash Check  Di Linux #
    #------------------------------------#
    clear #Untuk Membersihkan Layar Terminal Saat Membuka Script ini

    function check()
    {
        echo -n "Tool Yang Di Cari : "
        read tool
        echo -n "$tool : "
        if ! hash $tool 2>/dev/null
            then
                echo "Not Found"
                sleep 2
                check
            else
                echo "Found"
                echo "$tool In $(which $tool)"
                sleep 1
                check
        fi
    }
    check
